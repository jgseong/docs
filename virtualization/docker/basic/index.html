Docker

Docker Architecture

How to install

  • refer to Get Docker CE for Ubuntu

CLI

show docker container

# show running docker containers
docker ps 

# show all docker containers
docker ps -a

show docker image

docker images

search image

docker search 'image name'

pull image

docker pull 'image name'
  • examples
docker pull nginx
docker pull jekyll/minimal

launch container

docker run [options] --name 'container name' 'image name'
  • options
  -d : daemon mode
    -p : port forwarding
    --name : specify container name
  • example
docker run -it --name minilinux apline
docker run -d -p 8080:80 --name webserver nginx

control container

docker [start | restart | stop] webserver

delete container

docker rm [options] 'container name'
  • options
  -f : force

delete image

docker rmi 'container name'
